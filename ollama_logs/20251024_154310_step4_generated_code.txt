# Locate the login form
try:
    login_form = WebDriverWait(driver, 10).until(
        EC.presence_of_element_located((By.CSS_SELECTOR, "form[name='login']"))
    )
    print("Login form located.")
except Exception as e:
    print(f"Error locating login form: {e}")

# Fill in Email Address
try:
    email_input = login_form.find_element(By.ID, "input-4")
    email_input.clear()
    email_input.send_keys(resume_data.get("email", ""))
    print("Email address filled.")
except Exception as e:
    print(f"Error filling email address: {e}")

# Fill in Password
try:
    password_input = login_form.find_element(By.ID, "input-5")
    password_input.clear()
    password_input.send_keys(resume_data.get("password", ""))
    print("Password filled.")
except Exception as e:
    print(f"Error filling password: {e}")
