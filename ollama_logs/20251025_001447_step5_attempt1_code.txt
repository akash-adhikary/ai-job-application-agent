wait = WebDriverWait(driver, 15)
clicked = False

# Strategy 1: direct button with exact text
try:
    btn = wait.until(EC.element_to_be_clickable(
        (By.XPATH, "//button[normalize-space(text())='Create Account']")))
    driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", btn)
    time.sleep(1)
    driver.execute_script("arguments[0].click();", btn)
    print("✓ Clicked 'Create Account' button (direct XPath)")
    clicked = True
except Exception as e:
    print(f"Strategy 1 failed: {e}")

# Strategy 2: button containing the text
if not clicked:
    try:
        btn = wait.until(EC.element_to_be_clickable(
            (By.XPATH, "//button[contains(., 'Create Account')]")))
        driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", btn)
        time.sleep(1)
        try:
            btn.click()
            print("✓ Clicked 'Create Account' button (normal click)")
        except Exception:
            driver.execute_script("arguments[0].click();", btn)
            print("✓ Clicked 'Create Account' button (JS click)")
        clicked = True
    except Exception as e:
        print(f"Strategy 2 failed: {e}")

# Strategy 3: CSS selector by class (if unique)
if not clicked:
    try:
        btn = wait.until(EC.element_to_be_clickable(
            (By.CSS_SELECTOR, "button.css-1ru62dj")))
        driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", btn)
        time.sleep(1)
        driver.execute_script("arguments[0].click();", btn)
        print("✓ Clicked 'Create Account' button (CSS selector)")
        clicked = True
    except Exception as e:
        print(f"Strategy 3 failed: {e}")

# Strategy 4: fallback generic search
if not clicked:
    try:
        element = driver.find_element(
            By.XPATH, "//*[contains(text(),'Create Account')][1]")
        driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", element)
        time.sleep(1)
        driver.execute_script("arguments[0].click();", element)
        print("✓ Clicked element by generic text search")
        clicked = True
    except Exception as e:
        print(f"Strategy 4 failed: {e}")

if not clicked:
    print("✗ Could not click 'Create Account' button with any strategy")
