wait = WebDriverWait(driver, 15)

try:
    # Locate the password input inside the sign‑in modal
    password_input = wait.until(
        EC.element_to_be_clickable((By.ID, "input-5"))
    )
    driver.execute_script(
        "arguments[0].scrollIntoView({block: 'center'});", password_input
    )
    time.sleep(0.5)

    # Fill the password from resume_data
    password_input.clear()
    password_input.send_keys(resume_data["password"])
    print("✓ Filled password field")
    time.sleep(1)

except Exception as e:
    print(f"✗ Could not fill password field: {e}")
