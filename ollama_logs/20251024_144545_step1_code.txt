try:
    accept_btn = WebDriverWait(driver, 15).until(
        EC.element_to_be_clickable((By.XPATH, "//button[contains(text(),'Accept Cookies')]"))
    )
    driver.execute_script("arguments[0].scrollIntoView(true);", accept_btn)
    accept_btn.click()
    print("✓ Accepted cookies")
except Exception as e:
    print(f"⚠️ Could not accept cookies: {e}")
