try:
    wait = WebDriverWait(driver, 15)
    sign_in_btn = wait.until(
        EC.element_to_be_clickable(
            (By.XPATH, "//button[@type='submit' and contains(@class, 'css-16vlq4s')]")
        )
    )
    driver.execute_script("arguments[0].scrollIntoView(true);", sign_in_btn)
    sign_in_btn.click()
    print("✓ Clicked 'Sign In' button (class=css-16vlq4s)")
except Exception as e:
    print(f"✗ Failed to click 'Sign In' button: {e}")
