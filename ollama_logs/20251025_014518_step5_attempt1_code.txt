wait = WebDriverWait(driver, 15)
clicked = False

# Strategy 1: button with known class
try:
    btn = wait.until(EC.element_to_be_clickable((By.CSS_SELECTOR, "button.css-1ru62dj")))
    driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", btn)
    time.sleep(1)
    driver.execute_script("arguments[0].click();", btn)
    print("✓ Clicked 'Create Account' button (CSS class, JavaScript)")
    clicked = True
except Exception as e:
    print(f"Strategy 1 failed: {e}")

# Strategy 2: button with exact visible text
if not clicked:
    try:
        btn = wait.until(EC.element_to_be_clickable((By.XPATH, "//button[normalize-space()='Create Account']")))
        driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", btn)
        time.sleep(1)
        try:
            btn.click()
            print("✓ Clicked 'Create Account' button (normal click)")
        except Exception:
            driver.execute_script("arguments[0].click();", btn)
            print("✓ Clicked 'Create Account' button (JavaScript fallback)")
        clicked = True
    except Exception as e:
        print(f"Strategy 2 failed: {e}")

# Strategy 3: any element containing the text
if not clicked:
    try:
        btn = wait.until(EC.element_to_be_clickable((By.XPATH, "//*[contains(text(),'Create Account')][1]")))
        driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", btn)
        time.sleep(1)
        driver.execute_script("arguments[0].click();", btn)
        print("✓ Clicked element containing 'Create Account' (JavaScript)")
        clicked = True
    except Exception as e:
        print(f"Strategy 3 failed: {e}")

if not clicked:
    print("✗ Unable to locate and click the 'Create Account' button.")
