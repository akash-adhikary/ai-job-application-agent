wait = WebDriverWait(driver, 15)
try:
    # Locate the visible "Sign In" submit button below the password field
    sign_in_btn = wait.until(
        EC.element_to_be_clickable(
            (By.XPATH, "//button[normalize-space()='Sign In' and @type='submit']")
        )
    )
    driver.execute_script("arguments[0].scrollIntoView(true);", sign_in_btn)
    sign_in_btn.click()
    print("✓ Clicked Sign In button")
    time.sleep(1)
except Exception as e:
    print(f"✗ Failed to click Sign In button: {e}")
