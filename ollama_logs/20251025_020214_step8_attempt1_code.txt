from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.common.exceptions import ElementClickInterceptedException, TimeoutException
import time

wait = WebDriverWait(driver, 15)

try:
    # Primary selector: button with specific class and exact text
    sign_in_btn = wait.until(
        EC.element_to_be_clickable(
            (By.XPATH, "//button[contains(@class, 'css-1ru62dj') and normalize-space()='Sign In']")
        )
    )
except TimeoutException:
    # Fallback: any element with the class that contains the text
    try:
        sign_in_btn = wait.until(
            EC.element_to_be_clickable(
                (By.XPATH, "//*[contains(@class, 'css-1ru62dj') and contains(text(), 'Sign In')]")
            )
        )
    except TimeoutException:
        sign_in_btn = None

if sign_in_btn:
    driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", sign_in_btn)
    time.sleep(1)  # allow any lazy‑load animations to settle
    try:
        sign_in_btn.click()
        print("✓ Clicked Sign In button (normal click)")
    except ElementClickInterceptedException:
        driver.execute_script("arguments[0].click();", sign_in_btn)
        print("✓ Clicked Sign In button (JS click)")
else:
    print("✗ Could not locate the Sign In button")
