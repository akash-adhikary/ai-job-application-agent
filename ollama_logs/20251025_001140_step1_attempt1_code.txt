wait = WebDriverWait(driver, 15)
clicked = False

# Strategy 1: button element with exact visible text
try:
    accept_btn = wait.until(
        EC.element_to_be_clickable((By.XPATH, "//button[normalize-space(text())='Accept Cookies']"))
    )
    driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", accept_btn)
    time.sleep(1)
    driver.execute_script("arguments[0].click();", accept_btn)
    print("✓ Clicked 'Accept Cookies' button (XPath text match)")
    clicked = True
except Exception as e:
    print(f"✗ Strategy 1 failed: {e}")

# Strategy 2: button with known class (css-1pnd5en) and containing text
if not clicked:
    try:
        accept_btn = wait.until(
            EC.element_to_be_clickable(
                (By.CSS_SELECTOR, "button.css-1pnd5en")
            )
        )
        # verify text contains 'Accept Cookies' to avoid other buttons with same class
        if "Accept Cookies" in accept_btn.text:
            driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", accept_btn)
            time.sleep(1)
            driver.execute_script("arguments[0].click();", accept_btn)
            print("✓ Clicked 'Accept Cookies' button (CSS class match)")
            clicked = True
        else:
            raise Exception("Button text does not match")
    except Exception as e:
        print(f"✗ Strategy 2 failed: {e}")

# Strategy 3: generic clickable element containing the phrase
if not clicked:
    try:
        accept_btn = wait.until(
            EC.element_to_be_clickable(
                (By.XPATH, "//*[contains(text(),'Accept Cookies') and (self::button or self::a or self::div)]")
            )
        )
        driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", accept_btn)
        time.sleep(1)
        driver.execute_script("arguments[0].click();", accept_btn)
        print("✓ Clicked 'Accept Cookies' button (generic text search)")
        clicked = True
    except Exception as e:
        print(f"✗ Strategy 3 failed: {e}")

if not clicked:
    print("⚠️ Unable to locate and click the 'Accept Cookies' button.")
