wait = WebDriverWait(driver, 15)

try:
    # Locate the "Accept Cookies" button by its visible text
    accept_btn = wait.until(
        EC.element_to_be_clickable(
            (By.XPATH, "//button[normalize-space(text())='Accept Cookies']")
        )
    )
    # Scroll into view (centered) and click via JavaScript
    driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", accept_btn)
    time.sleep(1)
    driver.execute_script("arguments[0].click();", accept_btn)
    print("✓ Clicked 'Accept Cookies' button (JavaScript)")
    time.sleep(1)
except Exception as e:
    print(f"✗ Failed to click 'Accept Cookies' button: {e}")
