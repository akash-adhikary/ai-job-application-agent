wait = WebDriverWait(driver, 15)

try:
    # Locate the Sign In button (type="submit") with visible text "Sign In"
    sign_in_btn = wait.until(
        EC.element_to_be_clickable(
            (By.XPATH, "//button[@type='submit' and normalize-space()='Sign In']")
        )
    )
    driver.execute_script("arguments[0].scrollIntoView(true);", sign_in_btn)
    time.sleep(1)  # pause after scrolling
    sign_in_btn.click()
    print("✓ Clicked 'Sign In' button")
except Exception as e:
    print(f"✗ Failed to click 'Sign In' button: {e}")

time.sleep(1)  # ensure any post‑click actions have time to start
