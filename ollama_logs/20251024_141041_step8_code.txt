old_url = driver.current_url

# Click the 'Autofill with Resume' link
try:
    autofill_link = wait.until(
        EC.element_to_be_clickable((By.XPATH, "//a[normalize-space(text())='Autofill with Resume']"))
    )
    driver.execute_script("arguments[0].scrollIntoView(true);", autofill_link)
    autofill_link.click()
    print("Clicked 'Autofill with Resume'")
except Exception as e:
    print(f"Failed to click 'Autofill with Resume': {e}")

# Wait for URL to change after clicking
try:
    wait.until(lambda d: d.current_url != old_url)
    print("URL changed after clicking 'Autofill with Resume'")
except Exception as e:
    print(f"URL did not change after clicking 'Autofill with Resume': {e}")
