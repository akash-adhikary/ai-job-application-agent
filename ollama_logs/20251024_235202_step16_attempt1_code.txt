# Fill the "How Did You Hear About Us?" field (id: source--source) with "LinkedIn"
try:
    wait = WebDriverWait(driver, 15)
    source_input = wait.until(
        EC.element_to_be_clickable((By.ID, "source--source"))
    )
    driver.execute_script("arguments[0].scrollIntoView(true);", source_input)
    time.sleep(0.5)

    # Clear any existing text (just in case) and enter the source
    source_input.clear()
    source_input.send_keys("LinkedIn")
    print("✓ Filled 'How Did You Hear About Us?' with 'LinkedIn'")
except Exception as e:
    print(f"✗ Could not fill 'How Did You Hear About Us?' field: {e}")

time.sleep(1)  # brief pause before next actions
