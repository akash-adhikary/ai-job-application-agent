try:
    apply_link = wait.until(EC.element_to_be_clickable((By.CSS_SELECTOR, "html > body > div#root > div > div > div:nth-of-type(3) > div > main > div > div > div:nth-of-type(4) > div > div > a")))
    driver.execute_script("arguments[0].scrollIntoView(true);", apply_link)
    apply_link.click()
    print("Clicked 'Apply Manually' link.")
except Exception as e:
    print(f"Failed to click 'Apply Manually' link: {e}")

old_url = driver.current_url
try:
    wait.until(lambda d: d.current_url != old_url)
    print("URL changed after clicking.")
except Exception as e:
    print(f"URL did not change: {e}")
