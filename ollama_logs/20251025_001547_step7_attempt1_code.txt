wait = WebDriverWait(driver, 15)
clicked = False

# Strategy 1: Direct CSS selector using known class
try:
    btn = wait.until(EC.element_to_be_clickable((By.CSS_SELECTOR, "button.css-1ru62dj")))
    driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", btn)
    time.sleep(0.5)
    driver.execute_script("arguments[0].click();", btn)
    print("✓ Clicked 'Create Account' via class selector")
    clicked = True
except Exception as e:
    print(f"Strategy 1 failed: {e}")

# Strategy 2: Button with exact visible text
if not clicked:
    try:
        btn = wait.until(EC.element_to_be_clickable((By.XPATH, "//button[normalize-space(text())='Create Account']")))
        driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", btn)
        time.sleep(0.5)
        try:
            btn.click()
            print("✓ Clicked 'Create Account' via normal click")
        except ElementClickInterceptedException:
            driver.execute_script("arguments[0].click();", btn)
            print("✓ Clicked 'Create Account' via JavaScript click")
        clicked = True
    except Exception as e:
        print(f"Strategy 2 failed: {e}")

# Strategy 3: Any clickable element containing the text
if not clicked:
    try:
        btn = wait.until(EC.element_to_be_clickable((By.XPATH, "//*[contains(text(),'Create Account')][1]")))
        driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", btn)
        time.sleep(0.5)
        driver.execute_script("arguments[0].click();", btn)
        print("✓ Clicked 'Create Account' via generic text selector")
        clicked = True
    except Exception as e:
        print(f"Strategy 3 failed: {e}")

if not clicked:
    print("✗ Could not click the 'Create Account' button after all strategies.")
