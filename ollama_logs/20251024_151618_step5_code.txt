try:
    create_btn = WebDriverWait(driver, 15).until(
        EC.element_to_be_clickable((By.XPATH, "//button[normalize-space()='Create Account']"))
    )
    driver.execute_script("arguments[0].scrollIntoView(true);", create_btn)
    create_btn.click()
    print("✓ Clicked 'Create Account' button")
except Exception as e:
    print(f"✗ Failed to click 'Create Account' button: {e}")

time.sleep(1)
