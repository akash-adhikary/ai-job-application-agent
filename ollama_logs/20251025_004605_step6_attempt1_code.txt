wait = WebDriverWait(driver, 15)
clicked = False

# Strategy 1: button with exact text
try:
    btn = wait.until(EC.element_to_be_clickable((By.XPATH, "//button[normalize-space()='Create Account']")))
    driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", btn)
    time.sleep(1)
    driver.execute_script("arguments[0].click();", btn)
    print("✓ Clicked 'Create Account' (XPath exact text)")
    clicked = True
except Exception as e:
    print(f"Strategy 1 failed: {e}")

# Strategy 2: button with class name
if not clicked:
    try:
        btn = wait.until(EC.element_to_be_clickable((By.CSS_SELECTOR, "button.css-1ru62dj")))
        driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", btn)
        time.sleep(1)
        driver.execute_script("arguments[0].click();", btn)
        print("✓ Clicked 'Create Account' (CSS class)")
        clicked = True
    except Exception as e:
        print(f"Strategy 2 failed: {e}")

# Strategy 3: generic clickable element containing the text
if not clicked:
    try:
        btn = wait.until(EC.element_to_be_clickable((By.XPATH, "//*[contains(text(),'Create Account')][1]")))
        driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", btn)
        time.sleep(1)
        driver.execute_script("arguments[0].click();", btn)
        print("✓ Clicked 'Create Account' (generic text search)")
        clicked = True
    except Exception as e:
        print(f"Strategy 3 failed: {e}")

if not clicked:
    print("✗ Could not click the 'Create Account' button.")
