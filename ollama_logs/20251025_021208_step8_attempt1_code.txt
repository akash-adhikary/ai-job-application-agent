from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
import time

wait = WebDriverWait(driver, 15)

try:
    # Locate the "Create Account" button by its class and visible text
    create_btn = wait.until(
        EC.element_to_be_clickable(
            (By.XPATH, "//button[contains(@class, 'css-1ru62dj') and normalize-space()='Create Account']")
        )
    )
    driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", create_btn)
    time.sleep(1)

    try:
        create_btn.click()
        print("✓ Clicked 'Create Account' (normal click)")
    except Exception:
        driver.execute_script("arguments[0].click();", create_btn)
        print("✓ Clicked 'Create Account' (JavaScript click)")

    time.sleep(2)  # allow any post‑click processing
except Exception as e:
    print(f"✗ Could not click 'Create Account' button: {e}")
