wait = WebDriverWait(driver, 15)

try:
    # Primary selector for Workday sign‑in button
    sign_in_btn = wait.until(
        EC.element_to_be_clickable((By.CSS_SELECTOR, "button[data-automation-id='signInSubmitButton']"))
    )
    driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", sign_in_btn)
    time.sleep(0.5)
    driver.execute_script("arguments[0].click();", sign_in_btn)
    print("✓ Clicked Sign In (primary selector)")
except Exception as e_primary:
    print(f"⚠️ Primary sign‑in click failed: {e_primary}")
    try:
        # Fallback to class‑based selector
        fallback_btn = wait.until(
            EC.element_to_be_clickable((By.CSS_SELECTOR, "button.css-1ru62dj"))
        )
        driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", fallback_btn)
        time.sleep(0.5)
        driver.execute_script("arguments[0].click();", fallback_btn)
        print("✓ Clicked Sign In (fallback selector)")
    except Exception as e_fallback:
        print(f"❌ Both sign‑in attempts failed: {e_fallback}")
